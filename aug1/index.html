Aug 1 Programming Problem

Test Integer:

Number of integers containing '1':



Programming Solution
function numberOfOnes(n) {
var count = 0;
var inc = 1;
while (inc <= n) {
const incStr = inc.toString();
for(var i = 0; i < incStr.length; i++) {
if(incStr[i] === '1') {
count++;
}
}
inc++;
}
return count;
}
