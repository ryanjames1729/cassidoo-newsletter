Programming Solution
function formatOutput(doorArray) {
let output = "";

for(let i = 0; i < doorArray.length; i++) {
if(doorArray[i]) {
output += "1 ";
} else {
output += "0 ";
}
}
return output;
}

function passDoors(doors, passes) {
let output = "";
let doorArray = [];

for(let i = 0; i < doors; i++) {
doorArray.push(true);
}
output += "Initial state: " + formatOutput(doorArray) + "
";

for(let i = 2; i <= passes+1; i++) {
for(let j = i; j <= doors; j += i) {
doorArray[j - 1] = !doorArray[j - 1];
}
output += "After pass " + (i-1) + ": " + formatOutput(doorArray) + "
";
}
return output;
}
