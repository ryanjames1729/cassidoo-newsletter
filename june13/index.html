Programming Solution
function extraVowels(str, n) {
  const vowels = "aeiouAEIOU";
  let output = "";
  [...str].forEach(letter => {
   if (vowels.includes(letter)) {
    for(let i = 0; i < n; i++) {
     output += letter;
    }
   }
   else{
    output += letter;
   }
  })
 return output;
}
